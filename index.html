// ─── Core: Generate Report (UPDATED - uses comma instead of &) ───────────── function generateReport() { const room = state.selectedRoom; if (!room) { alert("Please select a room first."); return; } const normalizedRoom = normalizeStr(room); const metaRows = (state.metaData || []).filter((r) => { let excelRoom = r["Room No."] || r["Room"] || r["RoomNo"] || r["Room Number"] || ""; excelRoom = normalizeStr(excelRoom); return excelRoom === normalizedRoom || excelRoom.includes(normalizedRoom) || normalizedRoom.includes(excelRoom) || excelRoom.replace(/[\s.-]/g, '') === normalizedRoom.replace(/[\s.-]/g, ''); }); if (metaRows.length === 0) { console.log("Available rooms:", [...new Set((state.metaData || []).map(r => r["Room No."] || r["Room"]))]); alert(`No data found for room: ${room}`); return; } // ─── MULTI-PROGRAMME SUPPORT ───────────────────────────────────── const programmesSet = new Set(); const subjectsSet = new Set(); const rollRangesSet = new Set(); const allStudents = []; for (const row of metaRows) { const prog = row["Prog. Name & Stream"] || row["Programme"] || row["Prog Name"] || ""; if (prog) programmesSet.add(prog); const subj = row["Subject"] || row["Subjects"] || ""; if (subj) subjectsSet.add(subj); const range = row["Roll No. Range"] || row["Roll No.Range"] || row["Roll Range"] || row["Roll_No._Range"] || row["Roll No."] || ""; if (range) rollRangesSet.add(range); const ranges = parseRollRange(range); if (ranges) { for (const r of ranges) { const valid = (state.rollData || []).filter((student) => { const rollStr = String(student["Roll Number"] || student["Roll No."] || "").trim(); const num = extractNum(rollStr); const prefix = (rollStr.match(/[A-Za-z]+/) || [""])[0]; return !isNaN(num) && num >= r.start && num <= r.end && normalizeStr(prefix) === normalizeStr(r.prefix); }); valid.forEach((s) => { const rollVal = String(s["Roll Number"] || s["Roll No."] || "").trim(); if (rollVal && !allStudents.includes(rollVal)) { allStudents.push(rollVal); } }); } } } const info = { room: room, academicYear: metaRows[0]["Academic Year"] || "2025-2026", examType: state.examType || "Final Exam", date: formatDate(metaRows[0]["Exam Date"] || metaRows[0]["Date"]) || "—", time: metaRows[0]["Exam Time"] || metaRows[0]["Time"] || "—", programme: Array.from(programmesSet).join(", "), // ← Changed to comma subjects: Array.from(subjectsSet).join(", "), // ← Changed to comma sem: metaRows[0]["Semester"] || metaRows[0]["Sem"] || "I", rollRange: Array.from(rollRangesSet).join(", "), // ← Changed to comma totalStudents: allStudents.length }; state.layoutType = "auditorium"; let idx = 0; const assignedRows = state.rowConfigs.map((cfg) => { const rowData = { left: [], middle: [], right: [] }; const sides = ["left", "middle", "right"]; for (const side of sides) { const seatCount = Number(cfg[side]) || 0; for (let s = 0; s < seatCount; s++) { rowData[side].push(idx < allStudents.length ? allStudents[idx++] : ""); } } return rowData; }); renderReport(info, assignedRows, allStudents); }
